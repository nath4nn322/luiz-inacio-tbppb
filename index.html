from PIL import Image, ImageDraw, ImageFont import matplotlib.pyplot as plt # Criar uma imagem em branco width, height = 1000, 600 img = Image.new('RGB', (width, height), color='white') draw = ImageDraw.Draw(img) # Definir título e texto title = "🌱 Dicotiledôneas vs. Eudicotiledôneas" text = """ 📌 Dicotiledôneas (conceito antigo) - Plantas com 2 cotilédones - Folhas com nervuras ramificadas - Raiz pivotante - Flores geralmente com 4 ou 5 pétalas - NÃO formam um grupo natural (misturam várias origens) - Ex: Feijão, rosa, manga (e outros) ✅ Eudicotiledôneas (classificação moderna) - Também têm 2 cotilédones - Folhas com nervuras ramificadas - Raiz pivotante - Flores geralmente com 4 ou 5 pétalas - Formam um grupo natural (mesmo ancestral) - Têm pólen tricolpado (com 3 sulcos) - Ex: Feijão, tomate, girassol, maçã 🎯 Ou seja: Eudicotiledôneas = "dicotiledôneas verdadeiras" """ # Fonte (você pode mudar se quiser) font_title = ImageFont.truetype("DejaVuSans-Bold.ttf", 32) font_text = ImageFont.truetype("DejaVuSans.ttf", 20) # Desenhar texto draw.text((20, 20), title, font=font_title, fill="darkgreen") draw.text((20, 80), text, font=font_text, fill="black") # Mostrar a imagem plt.figure(figsize=(12, 7)) plt.imshow(img) plt.axis('off') plt.show()
